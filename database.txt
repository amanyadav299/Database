CREATE TABLE Personal_Info(
first_name VARCHAR2(10),
middle_name VARCHAR2(10),
last_name VARCHAR2(10),
citizenship_no NUMBER(10) PRIMARY KEY,
age NUMBER(10),
gender VARCHAR2(5));

CREATE TABLE Contact_Info(
  contact_no NUMBER(10) PRIMARY KEY,
  tel_no NUMBER(10),
  email VARCHAR2(20),
  work_email VARCHAR2(20),
  address VARCHAR2(20)
);

CREATE TABLE Medical_Info(
name VARCHAR2(30),
patient_No NUMBER(10) PRIMARY KEY,
date_of_birth DATE,
blood_group VARCHAR2(3),
insurance_no NUMBER(10));

CREATE TABLE Experience_Info(
name VARCHAR2(20),
employee_ID NUMBER(10) PRIMARY KEY,
contact_no NUMBER(10),
profession  VARCHAR2(15),
Company VARCHAR2(20),
address VARCHAR2(20));
